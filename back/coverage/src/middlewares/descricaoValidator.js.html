const descricaoValidator = (req, res, next) => {
  const obj = req.body;
 
  if (!Object.keys(obj).includes('descricao')) {
    return res.status(404).json({ message: 'Campo descrição requerido' });
  }
 
  if (obj.descricao.length < 5) {
    return res.status(404).json({ message: 'Campo descrição precisar ter 5 ou mais caracteres' });
  }
 
  next(); 
};
 
module.exports = { descricaoValidator };
